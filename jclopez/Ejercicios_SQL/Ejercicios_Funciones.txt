EJERCICIOS DE FUNCIONES

Realice una función que muestre los siguientes mensajes de acuerdo a la expectativa de vida:

Poca    :  exp <= 50
Media  :  exp > 50 && exp <= 60
Normal :  exp >= 60 && exp <= 68
Buena  : exp > 68

CREATE DEFINER = 'root'@'localhost' FUNCTION `FN_EXPECTATIVA`(`edad` FLOAT)
    RETURNS VARCHAR(50) CHARACTER SET latin1
    NOT DETERMINISTIC
    CONTAINS SQL
    SQL SECURITY DEFINER
    COMMENT ''
BEGIN
DECLARE RES VARCHAR(50);

      IF (edad <= 50) THEN
          SET RES = 'POCA';
      ELSE
          IF (edad > 50 and edad <= 60) THEN
                SET RES = 'MEDIA';
      ELSE
              IF (edad >= 60 and edad <=68) THEN
                   SET RES = 'NORMAL';
      ELSE
                  IF (edad > 68) THEN
                        SET RES = 'BUENA';
      			  END IF;
	  		  END IF;
     	  END IF;
      END IF;
RETURN RES;    
END;

Ejercicio 1

Realice una función que reciba el código de un país y diga si es un buen país para vivir, 
un buen país para vivir tiene expectativa de vida de más de 70 años, 
su PIB per cápita está por encima de los $10.000 dólares, tiene más de 20.000 kilómetros cuadrados 
y se hablan máximo 2 idiomas.

SOL:

CREATE FUNCTION `FN_BUEN_VIVIR`(`xcodigo` VARCHAR(3))
    RETURNS VARCHAR(50)
    NOT DETERMINISTIC
    CONTAINS SQL
    SQL SECURITY DEFINER
    COMMENT ''
BEGIN
DECLARE RES VARCHAR(50);
 SELECT 
   if((`country`.`LifeExpectancy` > 70) AND (`country`.`GNP` >= 10000) AND
      (`country`.`SurfaceArea` > 20000) AND 
      (SELECT  COUNT(`countrylanguage`.`IsOfficial`) AS `Idiomas`
	   FROM `countrylanguage` 
	   WHERE `countrylanguage`.`CountryCode`  = `country`.`Code` AND
        `countrylanguage`.`IsOfficial`='T'
	   GROUP BY `countrylanguage`.`CountryCode`
	   HAVING COUNT(`countrylanguage`.`IsOfficial`) = 2)
            , "BUEN_VIVIDERO", "MAL_VIVIDERO") INTO RES
FROM country
WHERE 	`country`.`Code`=xcodigo;
  RETURN RES;
END;


SELECT `FN_BUEN_VIVIR`('CAN')


Ejercicio 2

Realice una función que reciba el código de un país y muestre un resumen en texto de dicho país, ejemplo :
Para el código COL debe retornar :
Colombia es un país situado en Suramérica, tiene una población de 42,321,000 habitantes y una extensión 
de 1,138,914 de Km cuadrados, tiene una expectativa de vida normal (70 años) y se independizó en el año 
de 1810.

CREATE DEFINER = 'root'@'localhost' FUNCTION `FN_CONCATENAR`(
        `xcodigo` VARCHAR(30)
    )
    RETURNS VARCHAR(500) CHARACTER SET latin1
    NOT DETERMINISTIC
    CONTAINS SQL
    SQL SECURITY DEFINER
    COMMENT ''
BEGIN
DECLARE RES VARCHAR(500);
 SELECT 
CONCAT(`country`.`Name`,' es un pais situado en ', `country`.`Continent`,' tiene una poblacion de ',
 		FORMAT(`country`.`Population`,0),' habitantes y una extension de ',FORMAT(`country`.`SurfaceArea`,0),
        ' de Km cuadrados, tiene una expectativa de vida normal (',`country`.`LifeExpectancy`,' años) y se independizo en el año de ',
        `country`.`IndepYear`
       ) INTO RES
FROM country
WHERE 	`country`.`Code`=xcodigo;
  RETURN RES;
END;

 select `FN_CONCATENAR`('CAN');
